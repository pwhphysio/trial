─── Scheduling Algorithm ──────────────────────────────────── import math def auto_schedule(date_str): """ Auto-generate a schedule for the given date. Priority order: 1) Sports rooms: assign 1 capable staff per slot with class (check staffCapabilities). Half-day constraint applies. 2) Rm 4 + Private: default staff, substitute from pool if absent. 3) Rm 3/5/7: PCA allocation proportional to therapist count. Rm 3: 1 PCA per 3 therapists, Rm 5: 1 PCA per 2, Rm 7: 1 PCA per 2. If not enough remaining staff, distribute proportionally. """ dt = datetime.strptime(date_str, "%Y-%m-%d") day_idx = dt.weekday() if day_idx >= 5: return None # Weekend day_name = DAYS[day_idx] # Separate full-time and part-time staff full_time_staff = [s for s in state["staff"] if not s.get("partTime")] part_time_staff = [s for s in state["staff"] if s.get("partTime")] ft_names = [s["name"] for s in full_time_staff] pt_names = [s["name"] for s in part_time_staff] all_names = ft_names + pt_names # Rotate full-time staff list based on date so assignments are fair across days rotation_offset = dt.timetuple().tm_yday % len(ft_names) if ft_names else 0 rotated_names = ft_names[rotation_offset:] + ft_names[:rotation_offset] # Part-time staff only available if marked present pt_attendance = state.get("partTimeAttendance", {}).get(date_str, {}) date_abs = state["absences"].get(date_str, {}) day_tt = state["timetable"].get(day_name, {}) # Per-date private usage takes priority; fall back to weekly default day_priv_by_date = state.get("privateUsageByDate", {}).get(date_str, {}) day_priv_weekly = state["privateUsage"].get(day_name, {}) day_priv = day_priv_by_date if day_priv_by_date else day_priv_weekly follow_ups = state.get("followUps", {}).get(date_str, {}) day_tc = state["therapistCounts"].get(date_str, {}) pca_ratios = state.get("pcaRatios", {"Rm 3": 3, "Rm 5": 2, "Rm 7": 2}) capabilities = state.get("staffCapabilities", {}) def available(name, period): # Part-time staff: only available if explicitly marked present if name in pt_names: att = pt_attendance.get(name, {}) if att.get("fullday"): return True if period == "morning" and att.get("morning"): return True if period == "afternoon" and att.get("afternoon"): return True return False # part-time not marked present → not available # Full-time staff: available unless absent ab = date_abs.get(name, {}) if ab.get("fullday"): return False if period == "morning" and ab.get("morning"): return False if period == "afternoon" and ab.get("afternoon"): return False return True def fu_blocked(name, slot): """Check if staff is blocked by follow-up during this slot.""" fu_str = follow_ups.get(name, "") if not fu_str: return False fu_slots = [s.strip() for s in fu_str.split(",") if s.strip()] return slot in fu_slots no_pca_classes = state.get("noPcaClasses", []) def is_no_pca_class(class_name): """Check if this class doesn't need PCA (e.g. SCOLI, SAHK, Heep Hong).""" upper = class_name.upper() for npc in no_pca_classes: if npc.upper() in upper or upper in npc.upper(): return True return False def capable_of(name, class_name): """Check if staff is capable of the given sports class.""" if not capabilities: return True # No capability data → anyone can do anything cap_list = capabilities.get(class_name, []) if not cap_list: return True # No restriction for this class return name in cap_list sched = {room: {slot: [] for slot in TIME_SLOTS} for room in ALL_ROOMS} taken = {slot: set() for slot in TIME_SLOTS} sports_am = set() # staff in sports rooms during morning half sports_pm = set() # staff in sports rooms during afternoon half # ── Phase 1: Sports rooms (highest priority) ── # OAK1 and SNT1 classes require 2 staff members def sports_staff_needed(class_name): upper = class_name.upper() if upper.startswith("OAK1") or upper.startswith("SNT1"): return 2 return 1 # Rm 6 priority list rm6_priority = state["roomPriority"].get("Rm 6", []) for slot in TIME_SLOTS: period = "morning" if slot in MORNING_SLOTS else "afternoon" for room in SPORTS_ROOMS: cls = day_tt.get(room, {}).get(slot, "") if not cls: continue # Skip classes that don't need PCA if is_no_pca_class(cls): sched[room][slot] = [] continue needed = sports_staff_needed(cls) assigned_sports = [] # For Rm 6, try priority list first (e.g. Selina) if room == "Rm 6" and rm6_priority: for n in rm6_priority: if len(assigned_sports) >= needed: break if n in taken[slot]: continue if not available(n, period): continue if fu_blocked(n, slot): continue if not capable_of(n, cls): continue if period == "morning" and n in sports_pm: continue if period == "afternoon" and n in sports_am: continue assigned_sports.append(n) taken[slot].add(n) if period == "morning": sports_am.add(n) else: sports_pm.add(n) # Then fill from rotated pool (includes part-time if present) all_candidates = rotated_names + [n for n in pt_names if n not in rotated_names] for n in all_candidates: if len(assigned_sports) >= needed: break if n in taken[slot]: continue if n in assigned_sports: continue if not available(n, period): continue if fu_blocked(n, slot): continue if not capable_of(n, cls): continue # Half-day constraint if period == "morning" and n in sports_pm: continue if period == "afternoon" and n in sports_am: continue assigned_sports.append(n) taken[slot].add(n) if period == "morning": sports_am.add(n) else: sports_pm.add(n) sched[room][slot] = assigned_sports # ── Phase 2: Rm 4 + Private ── for slot in TIME_SLOTS: period = "morning" if slot in MORNING_SLOTS else "afternoon" # Rm 4 (default: Anit, Christy) cfg4 = state["roomDefaults"].get("Rm 4", {"staff": ["Anit", "Christy"], "count": 2}) a4 = [] for n in cfg4["staff"]: if len(a4) >= cfg4["count"]: break if n in all_names and available(n, period) and n not in taken[slot] and not fu_blocked(n, slot): a4.append(n) taken[slot].add(n) for n in rotated_names: if len(a4) >= cfg4["count"]: break if n not in a4 and available(n, period) and n not in taken[slot] and not fu_blocked(n, slot): a4.append(n) taken[slot].add(n) sched["Rm 4"][slot] = a4 # Private (default: Cat, only if in-use) if day_priv.get(slot, False): cfgP = state["roomDefaults"].get("Private", {"staff": ["Cat"], "count": 1}) aP = [] for n in cfgP["staff"]: if len(aP) >= cfgP["count"]: break if n in all_names and available(n, period) and n not in taken[slot] and not fu_blocked(n, slot): aP.append(n) taken[slot].add(n) for n in rotated_names: if len(aP) >= cfgP["count"]: break if n not in aP and available(n, period) and n not in taken[slot] and not fu_blocked(n, slot): aP.append(n) taken[slot].add(n) sched["Private"][slot] = aP # ── Phase 3: Rm 3/5/7 - PCA proportional allocation ── for slot in TIME_SLOTS: period = "morning" if slot in MORNING_SLOTS else "afternoon" # Calculate ideal PCA count per room pca_needs = {} # {room: ideal_count} for room in ["Rm 3", "Rm 5", "Rm 7"]: tc = day_tc.get(room, {}).get(slot, 0) ratio = pca_ratios.get(room, 2) if tc > 0 and ratio > 0: pca_needs[room] = math.ceil(tc / ratio) else: pca_needs[room] = 0 total_needed = sum(pca_needs.values()) # Count available staff for this slot avail_for_pca = [] for n in rotated_names: if n in taken[slot]: continue if not available(n, period): continue if fu_blocked(n, slot): continue avail_for_pca.append(n) # If enough staff, allocate as needed; otherwise proportionally reduce if len(avail_for_pca) >= total_needed: alloc = dict(pca_needs) elif total_needed > 0: # Proportional reduction alloc = {} remaining = len(avail_for_pca) for room in ["Rm 3", "Rm 5", "Rm 7"]: share = pca_needs[room] / total_needed * len(avail_for_pca) alloc[room] = int(share) # Distribute rounding remainders allocated_sum = sum(alloc.values()) leftover = len(avail_for_pca) - allocated_sum # Give leftovers to rooms with highest fractional part fracs = [] for room in ["Rm 3", "Rm 5", "Rm 7"]: share = pca_needs[room] / total_needed * len(avail_for_pca) fracs.append((share - int(share), room)) fracs.sort(reverse=True) for _, room in fracs: if leftover <= 0: break alloc[room] += 1 leftover -= 1 else: # No therapist counts set → distribute remaining staff evenly if avail_for_pca: base = len(avail_for_pca) // 3 extra = len(avail_for_pca) % 3 rooms_list = ["Rm 3", "Rm 5", "Rm 7"] alloc = {} for i, room in enumerate(rooms_list): alloc[room] = base + (1 if i < extra else 0) else: alloc = {room: 0 for room in ["Rm 3", "Rm 5", "Rm 7"]} # Assign staff using priority lists first, then general pool avail_set = set(avail_for_pca) for room in ["Rm 3", "Rm 5", "Rm 7"]: count = alloc[room] assigned = [] plist = state["roomPriority"].get(room, []) # Rotate priority list by same offset for fairness if plist: p_off = rotation_offset % len(plist) plist = plist[p_off:] + plist[:p_off] # Priority list first for n in plist: if len(assigned) >= count: break if n in avail_set: assigned.append(n) avail_set.discard(n) taken[slot].add(n) # Fallback to general pool if len(assigned) < count: for n in avail_for_pca: if len(assigned) >= count: break if n in avail_set: assigned.append(n) avail_set.discard(n) taken[slot].add(n) sched[room][slot] = assigned # ── Phase 4: Distribute remaining unassigned staff to Rm 3/5/7 ── for slot in TIME_SLOTS: period = "morning" if slot in MORNING_SLOTS else "afternoon" remaining = [] for n in rotated_names: if n in taken[slot]: continue if not available(n, period): continue if fu_blocked(n, slot): continue remaining.append(n) if remaining: # Round-robin distribute to Rm 3, Rm 5, Rm 7 target_rooms = ["Rm 3", "Rm 5", "Rm 7"] for i, n in enumerate(remaining): room = target_rooms[i % 3] sched[room][slot].append(n) taken[slot].add(n) state["schedules"][date_str] = sched save_data() return sched #
